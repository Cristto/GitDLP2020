expresión.tipo ∈ { TipoInt, TipoFloat, TipoChar }

PrintSp:Sentencia → expresion:Expresion

expresión.tipo ∈ { TipoInt, TipoFloat, TipoChar }

expIzq.modificable==true

If:Sentencia → condicion:Expresion sentenciasIf:Sentencia*

condicion.tipo==TipoInt

sentenciasi.funcion=If.funcion

IfElse:Sentencia → condicion:Expresion sentenciasIf:Sentencia* sentenciasElse:Sentencia*

condicion.tipo==TipoInt

sentenciasi.funcion=IfElse.funcion

While:Sentencia → condicion:Expresion sentencias:Sentencia*

condicion.tipo==TipoInt

sentenciasi.funcion=While.funcion

InvocacionProcedimiento:Sentencia → nombre:String argumentos:Expresion*

|InvocacionProcedimiento.argumentos|==|InvocacionProcedimiento.definicion.defParametros|

Return:Sentencia → expresion:Expresion

expresión.tipo ∈ { TipoInt, TipoFloat, TipoChar}

SI Return.funcion.tipo==null

      Return.tipo=null

SINO

     Return.tipo=Return.funcion.tipo

Read:Sentencia → expresion:Expresion

expresión.tipo ∈ { TipoInt, TipoFloat, TipoChar}

expresion.modificable==true

 

ExprAritmetica:Expresion → expIzq:Expresion operator:String expDrcha:Expresion

expIzq.tipo == expDrcha.tipo

expIzq.tipo ∈ { TipoInt, TipoFloat, TipoChar}

ExprAritmetica.tipo=exprIzq.tipo

ExprAritmetica.modificable=false

ExprComparacion:Expresion → expIzq:Expresion operator:String expDrcha:Expresion

expIzq.tipo == expDrcha.tipo

expIzq.tipo == TipoInt OR expIzq.tipo == TipoFloat

ExprComparacion.tipo=TipoInt

ExprAritmetica.modificable=false

ExprLogica:Expresion → expIzq:Expresion operator:String expDrcha:Expresion

expIzq.tipo==TipoInt

expDrcha.tipo==TipoInt

expIzq.tipo == expDrcha.tipo

ExprLogica.tipo=TipoInt

ExprAritmetica.modificable=false

ExprMenosUnario:Expresion → expr:ExpresionesTipoSimple(tipo)

expr.tipo ∈ { TipoInt, TipoFloat}

SI (expr.tipo==TipoChar)

    ExprMenosUnario.tipo=TipoInt

SINO

    ExprMenosUnario.tipo=expr.tipo

ExprAritmetica.modificable=false

ExprNot:Expresion → expr:Expresion

Expr.tipo==TipoInt

ExprNot.tipo=TipoInt

ExprAritmetica.modificable=false

AccesoArray:Expresion → identificador:Expresion posicion:Expresion

posicion.tipo==TipoInt

AccesoArray.modificable=true

ExpresionCast:Expresion → tipoCast:Tipo expresion:Expresion

expresion.tipo ∈ { TipoInt, TipoFloat, TipoChar}

tipoCast ∈ { TipoInt, TipoFloat, TipoChar}

expresion.tipo≠tipoCast

ExpresionCast.tipo=tipoCast

ExpresionModificable=false

AccesoStruct:Expresion → expresion:Expresion identificador:String

expresion.tipo==TipoStruct

AccesoStruct.modificable=true

ExpresionInvocacion:Expresion → nombre:String argumentos:Expresion*

|ExpresionInvocacion.argumentos|==|ExpresionInvocacion.definicion.DefParametros|

ExpresionInvocacion.argumentosi.tipo==ExpresionInvocacion.definicion.defParametrosi.tipo

ExpresionInvocacion.definicion.tipo≠null

ExpresionInvocacion.tipo= ExpresionInvocacion.definicion.tipo
